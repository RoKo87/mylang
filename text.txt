class Bread = {
    var size;
    var color;
    var thick;

    func bake() {
        size += 2;
        thick = true;
    }

    func stale() {
        size -= 2;
        thick = false;
    }

    constructor(s, t, c) {
        size = s;
        thick = t;
        color = c;
    }
}

var slice = new Bread(6, 5, "brown");
